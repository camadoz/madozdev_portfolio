                  for (var j = i + 1; j <= numbList.length - 1; j++)
                  {
                      if (numbList[i] + numbList[j] == kValue) {

                          clearSOFfields();
                          $("#sofAnswer").append($.parseHTML(kValue + " was found by adding array member " + numbList[i] + " and " + numbList[j] + ""));
                          foundSet = true;
                      }
                      //Exit all loops if set is found.
                      if (foundSet) break;
                  }
                }

                if (!foundSet)
                {
                    clearSOFfields();
                    $("#sofAnswer").append($.parseHTML("No set were found. Try again!"));
